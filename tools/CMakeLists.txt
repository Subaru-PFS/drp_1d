# Build all third party dependencies

  SET(SHARED_FLAG "--shared")
ELSE()
  SET(SHARED_FLAG "--static")
ENDIF()

  WORKING_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR}
  RESULT_VARIABLE result )
